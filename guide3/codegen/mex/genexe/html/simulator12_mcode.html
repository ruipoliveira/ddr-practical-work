1function [b o]= simulator1(lambda,invmiu,C,M,R)
 2	%lambda = request arrival rate (in requests per hour)
 3	%invmiu= average movie duration (in minutes)
 4	%C= Internet connection capacity (in Mbps)
 5	%M= throughput of each movie (in Mbps)
 6	%R= stop simulation on ARRIVAL no. R 
 7
 8	invlambda=60/lambda; %average time between requests (in minutes)
 9	
10	%Events definition:
11	ARRIVAL= 0; %movie request
12	DEPARTURE= 1; 	%termination of a movie transmission
13	
14	%State variables initialization:
15	STATE= 0;
16	
17	%Statistical counters initialization:
18	LOAD= 0;
19	NARRIVALS= 0;
20	BLOCKED= 0;
21	
22	%Simulation Clock and initial List of Events:
23	Clock= 0;
24	EventList= [ARRIVAL exprnd(invlambda)];
25	
26	while NARRIVALS < R
27		event= EventList(1,1);
28		Previous_Clock= Clock;
29		Clock= EventList(1,2);
30		EventList(1,:)= [];
31
32		LOAD= LOAD + STATE*(Clock-Previous_Clock);
33		
34		if event == ARRIVAL
35			EventList= [EventList; ARRIVAL Clock+exprnd(invlambda)];
36			NARRIVALS= NARRIVALS+1;
37			if STATE + M <= C
38				STATE= STATE+M;
39				EventList= [EventList; DEPARTURE Clock+exprnd(invmiu)];
40			else
41				BLOCKED= BLOCKED+1;
42			end
43		else
44			STATE= STATE-M;
45		end
46		EventList= sortrows(EventList,2);
47	end
48	
49	b= BLOCKED/NARRIVALS;
50	o= LOAD/Clock;
51
52end
